"""
contacts.py  ‚Äì CRUD CLI for JSON file
Commands:  add  list  search  delete
"""
import json, sys, pathlib

FILE = pathlib.Path("contacts.json")

def load():
    return json.loads(FILE.read_text()) if FILE.exists() else {}

def save(data):
    FILE.write_text(json.dumps(data, indent=2))

def add(name, phone):
    book = load()
    book[name] = phone
    save(book)
    print(f"Added {name}")

def list_all():
    for name, phone in load().items():
        print(f"{name:15} {phone}")

if __name__ == "__main__":
    cmd = sys.argv[1] if len(sys.argv) > 1 else ""
    match cmd:
        case "add":
            add(sys.argv[2], sys.argv[3])
        case "list":
            list_all()
        case _:
            print("Usage: python contacts.py add|list ‚Ä¶")

Slide 29  Decorators ‚Äì Function Wrappers

def debug(func):
    """Print call signature and return value."""
    import functools
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        args_repr = [repr(a) for a in args]
        kwargs_repr = [f"{k}={v!r}" for k, v in kwargs.items()]
        signature = ", ".join(args_repr + kwargs_repr)
        print(f"Calling {func.__name__}({signature})")
        value = func(*args, **kwargs)
        print(f"{func.__name__} returned {value!r}")
        return value
    return wrapper

@debug
def gcd(a, b):
    while b:
        a, b = b, a % b
    return a

gcd(42, 56)

def repeat(times: int):
    def decorator(func):
        import functools
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for _ in range(times):
                value = func(*args, **kwargs)
            return value
        return wrapper
    return decorator

@repeat(3)
def greet():
    print("Hi!")

import time, functools

def timed(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        t0 = time.perf_counter()
        value = func(*args, **kwargs)
        elapsed = time.perf_counter() - t0
        print(f"{func.__name__} took {elapsed*1000:.2f} ms")
        return value
    return wrapper

@functools.lru_cache(maxsize=None)
def fib(n):
    return n if n < 2 else fib(n-1) + fib(n-2)

timed(fib)(35))   # ~0.04 ms after cache warm

import threading, requests, time

urls = [f"https://picsum.photos/400/300?random={i}" for i in range(20)]

def download(url):
    resp = requests.get(url, timeout=10)
    filename = url.split("?")[-1] + ".jpg"
    with open(filename, "wb") as f:
        f.write(resp.content)
    print("done", filename)

t0 = time.time()
threads = [threading.Thread(target=download, args=(u,)) for u in urls]
for t in threads: t.start()
for t in threads: t.join()
print("Total:", time.time() - t0, "s")   # ~3 s instead of 30 s

Slide 36  multiprocessing ‚Äì CPU-bound Pi Monte-Carlo

import multiprocessing as mp, random

def sample_pi(n):
    """Return hits inside quarter-circle."""
    inside = 0
    for _ in range(n):
        x, y = random.random(), random.random()
        if x*x + y*y <= 1:
            inside += 1
    return inside

def main():
    cores = mp.cpu_count()
    total_samples = 1_000_000
    pool = mp.Pool(cores)
    chunk = total_samples // cores
    counts = pool.map(sample_pi, [chunk]*cores)
    pi_est = 4 * sum(counts) / total_samples
    print("œÄ ‚âà", pi_est)

if __name__ == "__main__":
    main()

import aiohttp, asyncio, time

async def fetch(session, url):
    async with session.get(url) as resp:
        return await resp.text()

async def fetch_all(urls):
    async with aiohttp.ClientSession() as session:
        tasks = [asyncio.create_task(fetch(session, u)) for u in urls]
        return await asyncio.gather(*tasks)

if __name__ == "__main__":
    urls = ["https://httpbin.org/delay/1"] * 50
    t0 = time.time()
    pages = asyncio.run(fetch_all(urls))
    print("Fetched", len(pages), "pages in", time.time()-t0, "s")
